Binary number To Octa Converter

Here User Input Is Binary Number As a String And The Output Of Program Is Octa Number. 
Java Code 

import java.util.Scanner;
public class Main {
    public static String BinToOcta(String binary){
            int Len1 = binary.length();
            String New_Bin = "";
            if (Len1 % 3 != 0) {
                if (Len1 % 3 == 2) {
                    New_Bin = "0" + binary;
                }
                if (Len1 % 3 == 1) {
                    New_Bin = "00" + binary;
                }
            }
            else {
                New_Bin = binary;
            }
            int Len = New_Bin.length();
            String NewString = "";
            int i=3;
            while(i<=Len){
                String TempNum = New_Bin.substring(Len-i,Len-i+3);
                int decimal = Integer.parseInt(TempNum, 2);
                NewString = decimal + NewString;
                i=i+3;
            }
            System.out.println(NewString);
            return NewString;
        }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        String binary = sc.nextLine();
        BinToOcta(binary);
    }
}
