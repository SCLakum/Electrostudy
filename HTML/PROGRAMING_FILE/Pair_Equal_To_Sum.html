import java.util.Scanner;
public class Main{
    public static void Pair_Equal_To_Sum(int ArrayLength, int Sum) {
        Scanner sc = new Scanner(System.in);
        System.out.println("Inter Your Integer Array Element One By one");
        int IntArray[] = new int[ArrayLength];
        for (int i = 0; i < ArrayLength; i++) {
            IntArray[i] = sc.nextInt();
        }
        int NumPair = 0;
        for (int i = 0; i<ArrayLength; i++) {
            int ArrayElem = IntArray[i];
            for (int j = 0; j<ArrayLength; j++) {
                if (j != i) {
                    if ((IntArray[i] + IntArray[j]) == Sum) {
                        if (IntArray[i]<IntArray[j]) {
                            System.out.println(IntArray[i] + " " + IntArray[j]);
                            NumPair++;
                        }
                    }
                }
            }
        }
        System.out.println("Number Of Pair =" + NumPair);
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter Length Of Array :");
        int arrLength = sc.nextInt();
        System.out.print("Enter Sum Of Pair :");
        int Sum = sc.nextInt();
        Pair_Equal_To_Sum(arrLength,Sum);
    }
