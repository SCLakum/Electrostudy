Volume Of Sphere

Here User Input Is Radius of Sphere And Output Is Volume Of Sphere. 
Java Code 

import java.util.Scanner;
public class Main{
    public static float Volume_Sphere(int Radius){
        float pi = 22f/7;
        float Volume = (4f/3)*pi*Radius*Radius*Radius; //  (4/3)Ï€r3
        return Volume;
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter Radius of Sphere:");
        int Radius = sc.nextInt();
        System.out.println("Volume :"+ Volume_Sphere(Radius));
    }
}
