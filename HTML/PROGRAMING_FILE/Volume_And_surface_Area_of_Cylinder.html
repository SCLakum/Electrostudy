Volume And Surface Area Of Cylinder

Here User Input Is Radius And Height Of Cylinder And Output Is Volume And Surface Area Of Cylinder. 
Java Code 

import java.util.Scanner;
public class Main{
    public static float Volume_Cylinder(int Radius, int Height){
        float pi = 22f/7;
        float Volume = pi*Radius*Radius*Height;
        return Volume;
    }
    public static float Surface_Area_Cylinder(int Radius, int Height){
        float pi = 22f/7;
        float Surface = 2*pi*Radius*Height + 2*pi*Radius*Radius;  // 2πrh+2πr2
        return Surface;
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter Radius Of cylinder:");
        int Radius = sc.nextInt();
        System.out.print("Enter Height  Of cylinder:");
        int Height = sc.nextInt();
        System.out.println("Volume :"+ Volume_Cylinder(Radius,Height));
        System.out.println("Surface Area :"+ Surface_Area_Cylinder(Radius,Height));
    }
}
